1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  export APP_HOST={{ $host.host }}
  echo "http://$APP_HOST"
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "openclaw.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  echo "Wait for the load balancer external IP/hostname to be assigned"
  kubectl get svc --namespace {{ .Release.Namespace }} {{ include "openclaw.fullname" . }} -w
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "openclaw.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT={{ .Values.service.targetPort }}
  echo "Visit http://127.0.0.1:18789 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 18789:$CONTAINER_PORT
{{- end }}
